#include<stdio.h>
int main() {
	int n,s,t,x;
	while(scanf("%d %d %d",&n,&s,&t)!=EOF) {
		x=0;
		if(s==t) {
			x=-1;
			if(n==1)
				x=0;
		} else {
			if((s==1&&t==n)||(s==n&&t==1))
				x=0;
			else if(s-t==1||s-t==-1) {
				x=1;
				if((s==1&&t!=2)||(s==n&&t!=n-1))
					x=0;
			} else {
				x=2;
				if(s==1||s==n)
					x=1;
			}
		}
		printf("%d\n",x);
	}
	return 0;
}
